This is what a template looks like:

<zine-wrapper>
  <zine-cover>
    <zine-left-page>
    </zine-left-page>
    <zine-right-page>
    </zine-right-page>
  </zine-cover>
  <zine-spread>
    <zine-left-page>
    </zine-left-page>
    <zine-right-page>
    </zine-right-page>
  </zine-spread>
  <zine-spread>
    <zine-left-page></zine-left-page>
    <zine-right-page></zine-right-page>
  </zine-spread>
  <zine-spread>
    <zine-left-page></zine-left-page>
    <zine-right-page></zine-right-page>
  </zine-spread>
</zine-wrapper>
